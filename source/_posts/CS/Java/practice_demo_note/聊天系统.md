### ClientChatMain

```java
package cn.chat;


import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.KeyEvent;
import java.awt.event.KeyListener;
import java.io.*;
import java.net.Socket;
import java.net.UnknownHostException;
import java.util.Properties;

public class ClientChatMain extends JFrame implements ActionListener, KeyListener {

    private static int clientPort;
    private static String clientIp;
    private BufferedWriter bw;
    private JTextArea jTextArea;
    private JScrollPane jScrollPane;
    private JPanel jPanel;
    private JTextField jTextField;
    private JButton jButton;

    //使用静态代码块加载配置文件内容
    static {
        Properties prop = new Properties();
        try {
            prop.load(new FileReader("src/cn/chat/resource/chat.properties"));
            clientPort=Integer.parseInt(prop.getProperty("clientPort"));
            clientIp=prop.getProperty("clientIp");
        } catch (IOException e) {
            e.printStackTrace();
        }
    }

    public ClientChatMain() {
        jTextArea = new JTextArea();
        jTextArea.setEditable(false);//文本域不可编辑
        jScrollPane = new JScrollPane(jTextArea);
        jPanel = new JPanel();
        jTextField = new JTextField(10);
        jButton = new JButton("send");
        jPanel.add(jTextField);
        jPanel.add(jButton);

        this.add(jScrollPane, BorderLayout.CENTER);
        this.add(jPanel, BorderLayout.SOUTH);

        this.setTitle("ChatSysterm-Client");
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setVisible(true);
        this.setSize(300, 300);
        this.setLocation(300, 300);

        /**** TCP Client start *****/
        //send 按钮的监听点击事件
        jButton.addActionListener(this);
        // 回车键监听事件
        jTextField.addKeyListener(this);
        try {
            // 1. 创建一个客户端的套接字（尝试连接）
            Socket socket = new Socket("127.0.0.1", 8888);
            // 2. 获取 socket 通道的输入流
            BufferedReader br = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            // 3. 获取 socket 通道的输出流
            bw = new BufferedWriter(new OutputStreamWriter(socket.getOutputStream()));

            // 循环读取数据,拼接到文本域中
            String line = null;
            while ((line = br.readLine()) != null) {
                jTextArea.append(line+'\n' );
            }

            // 4. 关闭 socket 通道
            socket.close();

        } catch (UnknownHostException e) {
            e.printStackTrace();
        } catch (IOException e) {
            e.printStackTrace();
        }
        /**** TCP Client end *****/

    }

    public static void main(String[] args) {

        System.out.println("hello");

        new ClientChatMain();

    }

    @Override
    public void actionPerformed(ActionEvent e) {
        sendToSocket();
    }

    private void sendToSocket() {
        //1. 获取文本框中发送的内容
        String text = jTextField.getText();
        //2. 拼接需要发送的数据内容
        text = "客户端 对 服务端说:" + text;
        jTextArea.append(text+'\n' );
        //3. 发送String str
        try {
            bw.write(text);
            bw.newLine();
            bw.flush();

            jTextField.setText("");
        } catch (IOException ex) {
            ex.printStackTrace();
        }
    }
    @Override
    public void keyTyped(KeyEvent e) {

    }

    @Override
    public void keyPressed(KeyEvent e) {
        if (e.getKeyCode() == 10) {
            sendToSocket();
        }
    }

    @Override
    public void keyReleased(KeyEvent e) {

    }
}
```

### ServerChatMain

```java
package cn.chat;

import javax.swing.*;
import java.awt.*;
import java.awt.event.ActionEvent;
import java.awt.event.ActionListener;
import java.awt.event.KeyEvent;
import java.awt.event.KeyListener;
import java.io.*;
import java.net.ServerSocket;
import java.net.Socket;
import java.util.Properties;

public class ServerChatMain extends JFrame implements ActionListener, KeyListener {

    private static int serverPort;
    private BufferedWriter bw=null;
    private JTextArea jTextArea;
    private JScrollPane jScrollPane;
    private JPanel jPanel;
    private JTextField jTextField;
    private JButton jButton;

    //使用静态代码块加载配置文件内容
    static {
        Properties prop = new Properties();
        try {
            prop.load(new FileReader("src/cn/chat/resource/chat.properties"));
            serverPort=Integer.parseInt(prop.getProperty("serverPort"));
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
    public ServerChatMain() {
        jTextArea = new JTextArea();
        jTextArea.setEditable(false);//文本域不可编辑
        jScrollPane = new JScrollPane(jTextArea);
        jPanel = new JPanel();
        jTextField = new JTextField(10);
        jButton = new JButton("send");
        jPanel.add(jTextField);
        jPanel.add(jButton);

        this.add(jScrollPane, BorderLayout.CENTER);
        this.add(jPanel, BorderLayout.SOUTH);

        this.setTitle("ChatSysterm-Server");
        this.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        this.setVisible(true);
        this.setSize(300, 300);
        this.setLocation(300, 300);

        /**** TCP Server start ******/
        //send 按钮的监听点击事件
        jButton.addActionListener(this);

        // 回车键监听事件
        jTextField.addKeyListener(this);
        try {
            // 1. 创建一个客户端的套接字
            ServerSocket serverSocket = new ServerSocket(8888);
            // 2. 等待客户端的连接
            Socket socket = serverSocket.accept();
            // 3. 获取 socket 通道的输入流
            BufferedReader br = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            // 4. 获取 socket 通道的输出流(写出数据,写一行,换一行,刷新)
            bw = new BufferedWriter(new OutputStreamWriter(socket.getOutputStream()));

            String line = null;
            while ((line = br.readLine()) != null) {
                jTextArea.append(line + '\n');
            }

            // 5. 关闭 socket 通道
            serverSocket.close();

        } catch (IOException e) {
            e.printStackTrace();
        }

        /**** TCP Server end ******/
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        sendToSocket();
    }

    public static void main(String[] args) {

        new ServerChatMain();

    }

    private void sendToSocket() {
        //1. 获取文本框中发送的内容
        String text = jTextField.getText();
        //2. 拼接需要发送的数据内容
        text = "服务端 对 客户端说:" + text;
        jTextArea.append(text+'\n' );
        //3. 发送String str
        try {
            if (bw != null) {
                bw.write(text);
            }
            bw.newLine();
            bw.flush();

            jTextField.setText("");
        } catch (IOException ex) {
            ex.printStackTrace();
        }
    }

    @Override
    public void keyTyped(KeyEvent e) {

    }

    @Override
    public void keyPressed(KeyEvent e) {
        if (e.getKeyCode() == 10) {
            sendToSocket();
        }
    }

    @Override
    public void keyReleased(KeyEvent e) {

    }
}
```
