预处理数组技巧示例 对于频繁需要查询的数据，可以先申请一个辅助的数据结构存储，下次需要使用数据时，直接从辅助数组中取，而不需要再进行遍历。 这种方法适用于某一步的查询是频繁的情况，但如果生成辅助数组本身耗时较长，则不适用。



/**
 * 染色问题：计算将一些正方形染成红色或绿色后，满足每个红色R都比每个绿色G距离最左侧近的最少染色次数。
 *
 * @param s 给定的正方形染色字符串，其中R表示红色，G表示绿色
 * @return 最少染色次数
 */
public static int minPaint(String s) {
	if (s == null || s.length() < 2) {
		return 0;
	}
	char[] chs = s.toCharArray();
	int[] right = new int[chs.length]; // 辅助数组，记录从当前位置到右边界的红色个数
	right[chs.length - 1] = chs[chs.length - 1] == 'R' ? 1 : 0;
	for (int i = chs.length - 2; i >= 0; i--) {
		right[i] = right[i + 1] + (chs[i] == 'R' ? 1 : 0);
	}
	int res = right[0];
	int left = 0;
	for (int i = 0; i < chs.length - 1; i++) {
		left += chs[i] == 'G' ? 1 : 0;
		res = Math.min(res, left + right[i + 1]);
	}
	res = Math.min(res, left + (chs[chs.length - 1] == 'G' ? 1 : 0));
	return res;
}

/**
 * 获取矩阵边框全是1的最大正方形的边长长度。
 *
 * @param m 给定的二维矩阵，只包含0和1，0表示白色，1表示黑色
 * @return 边框全是1的最大正方形的边长长度
 */
public static int getMaxSize(int[][] m) {
	int[][] right = new int[m.length][m[0].length]; // 辅助数组，记录从当前位置到右边界的连续1的个数
	int[][] down = new int[m.length][m[0].length]; // 辅助数组，记录从当前位置到下边界的连续1的个数
	setBorderMap(m, right, down); // 预处理辅助数组
	for (int size = Math.min(m.length, m[0].length); size != 0; size--) {
		if (hasSizeOfBorder(size, right, down)) {
			return size;
		}
	}
	return 0;
}

/**
 * 预处理辅助数组，计算每个位置到右边界和下边界的连续1的个数。
 *
 * @param m      给定的二维矩阵，只包含0和1，0表示白色，1表示黑色
 * @param right  用于记录从当前位置到右边界的连续1的个数的辅助数组
 * @param down   用于记录从当前位置到下边界的连续1的个数的辅助数组
 */
public static void setBorderMap(int[][] m, int[][] right, int[][] down) {
	int r = m.length;
	int c = m[0].length;
	if (m[r - 1][c - 1] == 1) {
		right[r - 1][c - 1] = 1;
		down[r - 1][c - 1] = 1;
	}
	for (int i = r - 2; i != -1; i--) {
		if (m[i][c - 1] == 1) {
			right[i][c - 1] = 1;
			down[i][c - 1] = down[i + 1][c - 1] + 1;
		}
	}
	for (int i = c - 2; i != -1; i--) {
		if (m[r - 1][i] == 1) {
			right[r - 1][i] = right[r - 1][i + 1] + 1;
			down[r - 1][i] = 1;
		}
	}
	for (int i = r - 2; i != -1; i--) {
		for (int j = c - 2; j != -1; j--) {
			if (m[i][j] == 1) {
				right[i][j] = right[i][j + 1] + 1;
				down[i][j] = down[i + 1][j] + 1;
			}
		}
	}
}

/**
 * 判断边长为size的正方形是否存在于矩阵的边框上。
 *
 * @param size   正方形的边长
 * @param right  用于记录从当前位置到右边界的连续1的个数的辅助数组
 * @param down   用于记录从当前位置到下边界的连续1的个数的辅助数组
 * @return 如果存在边长为size的正方形在边框上，则返回true，否则返回false
 */
public static boolean hasSizeOfBorder(int size, int[][] right, int[][] down) {
	for (int i = 0; i != right.length - size + 1; i++) {
		for (int j = 0; j != right[0].length - size + 1; j++) {
			if (right[i][j] >= size && down[i][j] >= size
					&& right[i + size - 1][j] >= size
					&& down[i][j + size - 1] >= size) {
				return true;
			}
		}
	}
	return false;
}

/**
 * 生成一个随机的0-1矩阵。
 *
 * @param rowSize    矩阵的行数
 * @param colSize

 矩阵的列数
 * @return 生成的随机0-1矩阵
 */
public static int[][] generateRandom01Matrix(int rowSize, int colSize) {
	int[][] res = new int[rowSize][colSize];
	for (int i = 0; i != rowSize; i++) {
		for (int j = 0; j != colSize; j++) {
			res[i][j] = (int) (Math.random() * 2);
		}
	}
	return res;
}

/**
 * 打印二维矩阵。
 *
 * @param matrix 要打印的二维矩阵
 */
public static void printMatrix(int[][] matrix) {
	for (int i = 0; i != matrix.length; i++) {
		for (int j = 0; j != matrix[0].length; j++) {
			System.out.print(matrix[i][j] + " ");
		}
		System.out.println();
	}
}
